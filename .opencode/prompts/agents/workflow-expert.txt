## Workflow Patterns from useworkflow.dev

### 1. Basic Workflow
```typescript
export async function exampleWorkflow(input: string) {
  "use workflow";
  const result = await processStep(input);
  return result;
}

async function processStep(input: string) {
  "use step";
  return { processed: input.toUpperCase() };
}
```

### 2. Multi-Step Workflow
```typescript
export async function onboardingWorkflow(user: User) {
  "use workflow";
  
  const account = await createAccountStep(user);
  const profile = await setupProfileStep(account);
  const welcome = await sendWelcomeStep(profile);
  
  return { account, profile, welcome };
}
```

### 3. Workflow with Conditional Logic
```typescript
export async function approvalWorkflow(request: Request) {
  "use workflow";
  
  const validation = await validateRequestStep(request);
  
  if (validation.needsReview) {
    const review = await reviewStep(request);
    return { status: 'pending_review', review };
  }
  
  const approved = await approveStep(request);
  return { status: 'approved', approved };
}
```

### 4. Workflow with Error Handling
```typescript
export async function robustWorkflow(input: Data) {
  "use workflow";
  
  try {
    const result = await riskyStep(input);
    return { success: true, result };
  } catch (error) {
    const fallback = await fallbackStep(input);
    return { success: false, fallback, error: String(error) };
  }
}
```

### 5. Parallel Workflow Steps
```typescript
export async function parallelWorkflow(ids: string[]) {
  "use workflow";
  
  const promises = ids.map(id => fetchItemStep(id));
  const results = await Promise.all(promises);
  
  return results;
}
```

## Step Function Best Practices

### Do
- Use descriptive names: `validateUserStep` not `step1`
- Handle errors within steps
- Return serializable data
- Keep steps focused

### Don't
- Don't call external APIs directly in workflows
- Don't mutate input parameters
- Don't rely on global state
- Don't use non-deterministic functions (Date.now(), Math.random())

## Error Handling Patterns

### Retry with Backoff
```typescript
async function resilientStep(input: Data) {
  "use step";
  
  let lastError;
  for (let attempt = 0; attempt < 3; attempt++) {
    try {
      return await callExternalAPI(input);
    } catch (error) {
      lastError = error;
      await sleep(Math.pow(2, attempt) * 1000); // 1s, 2s, 4s
    }
  }
  throw lastError;
}
```

### Circuit Breaker
```typescript
class CircuitBreaker {
  failures = 0;
  lastFailure = 0;
  
  async execute<T>(fn: () => Promise<T>): Promise<T> {
    if (this.isOpen()) throw new Error('Circuit open');
    
    try {
      const result = await fn();
      this.failures = 0;
      return result;
    } catch (error) {
      this.recordFailure();
      throw error;
    }
  }
  
  isOpen(): boolean {
    if (this.failures >= 5) {
      const elapsed = Date.now() - this.lastFailure;
      return elapsed < 60000; // 1 minute
    }
    return false;
  }
  
  recordFailure() {
    this.failures++;
    this.lastFailure = Date.now();
  }
}
```

## Performance Tips

1. **Batch operations** - Process multiple items in a single step
2. **Cache results** - Use workflow context for caching
3. **Lazy evaluation** - Only compute what's needed
4. **Parallel steps** - Use Promise.all for independent steps
