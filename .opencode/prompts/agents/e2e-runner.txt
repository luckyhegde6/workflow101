## E2E Test Structure

### 1. Page Object Model
```typescript
// tests/e2e/pages/DashboardPage.ts
export class DashboardPage {
  constructor(private page: Page) {}

  async goto() {
    await this.page.goto('/');
  }

  async getWorkflowCount(): Promise<number> {
    return this.page.locator('[data-testid="workflow-card"]').count();
  }

  async clickWorkflow(name: string) {
    await this.page.locator(`[data-testid="workflow-${name}"]`).click();
  }
}
```

### 2. Test Fixtures
```typescript
// tests/e2e/fixtures.ts
import { test as base } from '@playwright/test';

export const test = base.extend<{
  dashboardPage: DashboardPage;
}>({
  dashboardPage: async ({ page }, use) => {
    const dashboard = new DashboardPage(page);
    await use(dashboard);
  },
});
```

### 3. Critical User Flows

```typescript
// tests/e2e/dashboard.spec.ts
import { test, expect } from '@playwright/test';
import { test as base } from './fixtures';

const test = base.extend({ /* fixtures */ });

test.describe('Dashboard', () => {
  test('displays workflow list', async ({ dashboardPage }) => {
    await dashboardPage.goto();
    const count = await dashboardPage.getWorkflowCount();
    expect(count).toBeGreaterThan(0);
  });

  test('shows workflow details on click', async ({ dashboardPage }) => {
    await dashboardPage.goto();
    await dashboardPage.clickWorkflow('Example Workflow');
    await expect(page).toHaveURL(/\/workflows\/.+/);
  });
});
```

## Test Reliability

### Avoid Flaky Tests
```typescript
// BAD - Race condition
await page.click('#submit');
await page.waitForTimeout(1000);
await expect(page.locator('.success')).toBeVisible();

// GOOD - Wait for actual condition
await page.click('#submit');
await expect(page.locator('.success')).toBeVisible({ timeout: 5000 });
```

### Proper Waiting
```typescript
// Wait for network idle
await page.goto(url, { waitUntil: 'networkidle' });

// Wait for specific element
await page.waitForSelector('[data-testid="workflow-list"]');

// Wait for API response
await page.waitForResponse(resp => resp.url().includes('/api/workflows'));
```

## Running E2E Tests

```bash
# Run all E2E tests
npm run test:e2e

# Run specific test file
npm run test:e2e -- tests/e2e/dashboard.spec.ts

# Run with UI
npm run test:e2e -- --ui

# Run with headed browser
npm run test:e2e -- --headed

# Debug mode
npm run test:e2e -- --debug
```

## Debugging Failed Tests

```typescript
// Add debug info to tests
test('workflow execution', async ({ page }) => {
  // Take screenshot on failure
  await use.screenshotOnFailure();
  
  // Record video
  await use.video();
  
  // Trace viewer
  await page.context().tracing.start({ screenshots: true });
});
```
